-- Users table
create table public.users (
  id uuid references auth.users not null primary key,
  email text not null,
  name text,
  created_at timestamp with time zone default now() not null
);

-- Wineries table
create table public.wineries (
  id bigint primary key,
  name text not null,
  website text,
  address text,
  latitude decimal(10, 6),
  longitude decimal(10, 6),
  created_at timestamp with time zone default now() not null
);

-- Visits table
create table public.visits (
  id uuid default uuid_generate_v4() primary key,
  user_id uuid references public.users not null,
  winery_id bigint references public.wineries not null,
  visit_date date not null,
  notes text,
  photo_url text,
  created_at timestamp with time zone default now() not null
);

-- Wines table 
create table public.wines (
  id uuid default uuid_generate_v4() primary key,
  visit_id uuid references public.visits,
  wine_name text not null,
  wine_type text not null,
  wine_year text,
  overall_rating decimal(3, 1) not null,
  sweetness decimal(3, 1),
  tannin decimal(3, 1),
  acidity decimal(3, 1),
  body decimal(3, 1),
  alcohol decimal(3, 1),
  additional_notes text,
  photo_url text,
  created_at timestamp with time zone default now() not null
);

-- Flavor notes table (many-to-many)
create table public.flavor_notes (
  id uuid default uuid_generate_v4() primary key,
  name text not null unique,
  category text not null
);

-- Wine flavor notes (junction table)
create table public.wine_flavor_notes (
  id uuid default uuid_generate_v4() primary key,
  wine_id uuid references public.wines not null,
  flavor_note_id uuid references public.flavor_notes not null,
  unique(wine_id, flavor_note_id)
);

-- Enable RLS on all tables
alter table public.users enable row level security;
alter table public.wineries enable row level security;
alter table public.visits enable row level security;
alter table public.wines enable row level security;
alter table public.flavor_notes enable row level security;
alter table public.wine_flavor_notes enable row level security;

-- Create policies
-- Users can read all wineries
create policy "Anyone can read wineries" on public.wineries for select using (true);

-- Users can only read their own profile
create policy "Users can read their own profile" on public.users for select using (auth.uid() = id);

-- Users can create their own profile
create policy "Users can create their own profile" on public.users for insert with check (auth.uid() = id);

-- Users can update their own profile
create policy "Users can update their own profile" on public.users for update using (auth.uid() = id);

-- Users can read only their own visits
create policy "Users can read their own visits" on public.visits for select using (auth.uid() = user_id);

-- Users can insert only their own visits
create policy "Users can insert their own visits" on public.visits for insert with check (auth.uid() = user_id);

-- Users can update only their own visits
create policy "Users can update their own visits" on public.visits for update using (auth.uid() = user_id);

-- Users can delete only their own visits
create policy "Users can delete their own visits" on public.visits for delete using (auth.uid() = user_id);

-- Users can read wines associated with their visits
create policy "Users can read their own wines" on public.wines for select 
using (
  visit_id in (select id from public.visits where user_id = auth.uid())
);

-- Users can insert wines to their own visits
create policy "Users can insert wines to their own visits" on public.wines for insert 
with check (
  visit_id in (select id from public.visits where user_id = auth.uid())
);

-- Users can update wines from their own visits
create policy "Users can update their own wines" on public.wines for update 
using (
  visit_id in (select id from public.visits where user_id = auth.uid())
);

-- Users can delete wines from their own visits
create policy "Users can delete their own wines" on public.wines for delete 
using (
  visit_id in (select id from public.visits where user_id = auth.uid())
);

-- Anyone can read flavor notes
create policy "Anyone can read flavor notes" on public.flavor_notes for select using (true);

-- Users can read their own wine-flavor connections
create policy "Users can read their own wine flavor notes" on public.wine_flavor_notes for select 
using (
  wine_id in (
    select w.id from public.wines w
    join public.visits v on w.visit_id = v.id
    where v.user_id = auth.uid()
  )
);

-- Users can create wine-flavor connections for their own wines
create policy "Users can insert their own wine flavor notes" on public.wine_flavor_notes for insert 
with check (
  wine_id in (
    select w.id from public.wines w
    join public.visits v on w.visit_id = v.id
    where v.user_id = auth.uid()
  )
);

-- Users can delete wine-flavor connections for their own wines
create policy "Users can delete their own wine flavor notes" on public.wine_flavor_notes for delete 
using (
  wine_id in (
    select w.id from public.wines w
    join public.visits v on w.visit_id = v.id
    where v.user_id = auth.uid()
  )
);
